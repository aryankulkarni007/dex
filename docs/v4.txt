struct Point {
  x: flt, y: flt

  @(self, x, y) create -> Point {
    self.x = x
    self.y = y
    self
  }

  @(self, other) distance -> flt {
    ((self.x - other.x)^2 + (self.y - other.y)^2)^0.5
  }

  @(self) print -> abyss {
    !("({}, {})", self.x, self.y)
  }
}

@(list: [int]) double_evens -> [int] {
  list >> filter (n -> n % 2 == 0) >> map (n -> n * 2)
}
